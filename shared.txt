## RUN ON SERVER — force GPU, diagnose, and verify:

# 1. Pull latest fixes
git pull

# 2. Full reconfigure (fixes permissions, adds LD_LIBRARY_PATH, migrates models)
./scripts/setup-ollama.sh

# 3. Check CUDA runners and logs
./scripts/setup-ollama.sh check

# 4. THE DEFINITIVE TEST — loads a model and confirms "100% GPU" in ollama ps
./scripts/setup-ollama.sh test-gpu

# 5. Capture everything
{
  echo "=== OLLAMA LIST ==="
  ollama list
  echo "=== OLLAMA PS ==="
  ollama ps
  echo "=== CUDA RUNNERS ==="
  ls /usr/local/lib/ollama/runners/ 2>/dev/null || echo "dir not found"
  echo "=== GPU ==="
  nvidia-smi --query-gpu=index,name,memory.used,memory.total,utilization.gpu --format=csv,noheader
  echo "=== GPU PROCESSES ==="
  nvidia-smi --query-compute-apps=pid,name,used_gpu_memory --format=csv,noheader
  echo "=== OLLAMA SERVICE USER ==="
  systemctl show ollama --property=User
  echo "=== DATA/OLLAMA OWNER ==="
  ls -la /data/ollama/ | head -5
  echo "=== JOURNAL GPU LINES ==="
  journalctl -u ollama --no-pager -n 50 | grep -iE "cuda|gpu|runner|layer|cpu" | tail -20
} > /tmp/out.txt 2>&1
cat /tmp/out.txt
cp /tmp/out.txt shared.txt; git add shared.txt && git commit -m "gpu test output" && git push
