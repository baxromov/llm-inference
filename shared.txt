## RUN ON SERVER — stop old containers, then deploy with new host-ollama architecture:

docker compose down 2>/dev/null; git pull && ./deploy.sh 2>&1 | tee /tmp/out.txt; echo "=== OLLAMA STATUS ===" >> /tmp/out.txt; systemctl status ollama --no-pager | head -15 >> /tmp/out.txt; echo "=== OLLAMA GPU ===" >> /tmp/out.txt; ollama list 2>&1 >> /tmp/out.txt; echo "=== LITELLM ===" >> /tmp/out.txt; docker ps >> /tmp/out.txt; cp /tmp/out.txt shared.txt; git add shared.txt && git commit -m "deploy output" && git push
