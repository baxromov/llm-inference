## RUN ON SERVER — fix storage then deploy:

# 1. Clean up failed partial download (freeing space on /)
rm -rf /root/.ollama/models/blobs/*-partial-* 2>/dev/null; rm -rf /root/.ollama/models/blobs/sha256-f5* 2>/dev/null

# 2. Deploy — this configures OLLAMA_MODELS=/data/ollama/models and restarts service
git pull && ./deploy.sh 2>&1 | tee /tmp/out.txt

# 3. Capture status
echo "=== OLLAMA SERVICE ===" >> /tmp/out.txt
systemctl status ollama --no-pager | head -15 >> /tmp/out.txt
echo "=== OLLAMA ENV ===" >> /tmp/out.txt
systemctl show ollama --property=Environment >> /tmp/out.txt
echo "=== OLLAMA LIST ===" >> /tmp/out.txt
ollama list >> /tmp/out.txt
echo "=== DISK USAGE ===" >> /tmp/out.txt
df -h /data / >> /tmp/out.txt
echo "=== DOCKER ===" >> /tmp/out.txt
docker ps >> /tmp/out.txt
cp /tmp/out.txt shared.txt; git add shared.txt && git commit -m "deploy output" && git push
