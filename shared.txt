## RUN ON SERVER — fix missing models then verify GPU:

# 1. Pull latest fixes
git pull

# 2. Re-run setup — this will migrate /root/.ollama/models → /data/ollama automatically
./scripts/setup-ollama.sh

# 3. Capture status
echo "=== OLLAMA LIST ===" > /tmp/out.txt
ollama list >> /tmp/out.txt
echo "=== OLLAMA PS ===" >> /tmp/out.txt
ollama ps >> /tmp/out.txt
echo "=== GPU ===" >> /tmp/out.txt
nvidia-smi --query-gpu=index,name,memory.used,memory.total,utilization.gpu --format=csv,noheader >> /tmp/out.txt
echo "=== DISK /data ===" >> /tmp/out.txt
df -h /data >> /tmp/out.txt
echo "=== /data/ollama contents ===" >> /tmp/out.txt
ls -lh /data/ollama/ >> /tmp/out.txt 2>&1
cp /tmp/out.txt shared.txt; git add shared.txt && git commit -m "server status" && git push
