## RUN ON SERVER — root cause diagnostics:

# 1. Check gpt-oss Modelfile — look for "num_gpu 0"
ollama show gpt-oss --modelfile

# 2. Run with OLLAMA_DEBUG to see CUDA init messages
sudo systemctl stop ollama
sleep 2
sudo OLLAMA_DEBUG=1 OLLAMA_MODELS=/data/ollama CUDA_VISIBLE_DEVICES=all ollama serve > /tmp/ollama_debug.log 2>&1 &
sleep 5
# Load model and check GPU
curl -s http://localhost:11434/api/generate \
  -d '{"model":"gpt-oss","prompt":"hi","stream":false,"options":{"num_gpu":-1},"keep_alive":"60s"}' \
  > /tmp/gen_resp.json
ollama ps
# Stop debug server
kill %1 2>/dev/null; sudo systemctl start ollama

# 3. Show CUDA init lines from debug log
echo "=== CUDA INIT LINES ==="
grep -iE "cuda|gpu|ngl|layer|runner|ggml|rocm|metal|cpu" /tmp/ollama_debug.log | head -30

# Capture all
{
  echo "=== MODELFILE ==="
  ollama show gpt-oss --modelfile
  echo "=== CUDA RUNNERS ==="
  ls -la /usr/local/lib/ollama/runners/ 2>/dev/null
  echo "=== OLLAMA VERSION ==="
  ollama --version
  echo "=== CUDA INIT DEBUG ==="
  grep -iE "cuda|gpu|ngl|layer|runner|ggml|error" /tmp/ollama_debug.log | head -40
  echo "=== OLLAMA PS AFTER num_gpu=-1 ==="
  cat /tmp/gen_resp.json | python3 -c "import sys,json; d=json.load(sys.stdin); print('done')" 2>/dev/null
} > /tmp/out.txt 2>&1
cat /tmp/out.txt
cp /tmp/out.txt shared.txt; git add shared.txt && git commit -m "gpu root cause diagnostics" && git push
